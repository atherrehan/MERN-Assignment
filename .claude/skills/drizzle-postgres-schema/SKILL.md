# Drizzle + PostgreSQL Schema (`backend/src/db`)

The data layer for the Country & State app. Only **service classes** import from here
(see [express-api-architecture]). Types must match the hub contract [country-state-app].

## Folder structure

```
backend/src/db/
├─ schema.ts        # countries, states tables + relations
├─ client.ts        # drizzle client over a pg pool
├─ seed.ts          # idempotent dev seed
└─ migrations/      # generated by drizzle-kit
drizzle.config.ts   # at backend/ root
```

## Schema

Fields are **exactly** the entity contract — no extras. `id` unique; `State.countryId`
is a FK to `Country.id`.

```ts
// db/schema.ts
import { pgTable, serial, varchar, boolean, integer, index } from 'drizzle-orm/pg-core';
import { relations } from 'drizzle-orm';

export const countries = pgTable('countries', {
  id:       serial('id').primaryKey(),
  code:     varchar('code', { length: 8 }).notNull().unique(),
  name:     varchar('name', { length: 128 }).notNull(),
  isActive: boolean('is_active').notNull().default(true),
});

export const states = pgTable('states', {
  id:        serial('id').primaryKey(),
  code:      varchar('code', { length: 8 }).notNull(),
  name:      varchar('name', { length: 128 }).notNull(),
  isActive:  boolean('is_active').notNull().default(true),
  countryId: integer('country_id')
    .notNull()
    .references(() => countries.id, { onDelete: 'cascade' }),
}, (t) => ({
  countryIdx: index('states_country_id_idx').on(t.countryId),
}));

export const countriesRelations = relations(countries, ({ many }) => ({
  states: many(states),
}));
export const statesRelations = relations(states, ({ one }) => ({
  country: one(countries, { fields: [states.countryId], references: [countries.id] }),
}));
```

> The DB column is `is_active`; Drizzle maps it to the `isActive` field so rows match
> the shared `Country`/`State` types directly.

## DB client

```ts
// db/client.ts
import { drizzle } from 'drizzle-orm/node-postgres';
import { Pool } from 'pg';
import * as schema from './schema';
import { env } from '../config/env';

const pool = new Pool({ connectionString: env.DATABASE_URL });
export const db = drizzle(pool, { schema });
```

## drizzle-kit config

```ts
// drizzle.config.ts
import type { Config } from 'drizzle-kit';
export default {
  schema: './src/db/schema.ts',
  out: './src/db/migrations',
  dialect: 'postgresql',
  dbCredentials: { url: process.env.DATABASE_URL! },
} satisfies Config;
```

## Migration workflow

```bash
npx drizzle-kit generate   # create SQL from schema changes
npx drizzle-kit migrate    # apply migrations to DATABASE_URL
```

- Commit generated migrations. Never edit applied migration SQL — generate a new one.
- Run `migrate` on deploy before the API starts serving.

## Seed (idempotent)

```ts
// db/seed.ts
import { db } from './client';
import { countries, states } from './schema';

async function seed() {
  const [pk] = await db.insert(countries)
    .values({ code: 'PK', name: 'Pakistan', isActive: true })
    .onConflictDoNothing({ target: countries.code })
    .returning();
  if (pk) {
    await db.insert(states).values([
      { code: 'PB', name: 'Punjab',  countryId: pk.id, isActive: true },
      { code: 'SD', name: 'Sindh',   countryId: pk.id, isActive: true },
    ]).onConflictDoNothing();
  }
}
seed().then(() => process.exit(0)).catch((e) => { console.error(e); process.exit(1); });
```

## Search / filter query pattern (states)

Used by `StateService.list({ countryId?, search? })`. Build a conditions array and
combine with `and`; case-insensitive name match via `ilike`.

```ts
import { and, eq, ilike } from 'drizzle-orm';
import { db } from './client';
import { states } from './schema';

async function findStates(opts: { countryId?: number; search?: string }) {
  const conds = [];
  if (opts.countryId) conds.push(eq(states.countryId, opts.countryId));
  if (opts.search)    conds.push(ilike(states.name, `%${opts.search}%`));
  return db.select().from(states).where(conds.length ? and(...conds) : undefined);
}
```

## Rules

1. Table fields match the entity contract exactly — no extra columns.
2. `State.countryId` is a non-null FK to `Country.id` (`onDelete: cascade`).
3. Only services import `db/`. No raw SQL outside this layer.
4. Schema changes go through `drizzle-kit generate` → `migrate`; migrations are committed.
5. Returned rows conform to shared `Country`/`State` types.
